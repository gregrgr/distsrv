使用示例

# 安装 CLI（macOS）
curl -fsSL {{.BaseURL}}/static/mac/install.sh | bash

# 配置
distsrv-cli configure --server {{.BaseURL}} --token <your-token>

# 列出应用
distsrv-cli apps

# 上传 ipa / apk
distsrv-cli upload myapp ./build/MyApp.ipa
distsrv-cli upload myapp ./build/MyApp.apk

# 直接走 curl（无需 CLI）
curl -fsSL -H "Authorization: Bearer <your-token>" \
     -F "file=@./MyApp.ipa" \
     {{.BaseURL}}/api/v1/apps/myapp/upload
